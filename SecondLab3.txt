package SecondLab;
//write a program to input any three angles and check they form a triangle or not.
public class TriangleProgram {
	   public static void main(String[] args)
	     {
	    	 int a=80;
	    	 int b=20;
	    	 int c=80;
	    	 int d=a+b+c;
	    	 if(d==180)
	    	 {
	    		 System.out.println("It is a triangle");
	    	 }
	    	 else
	    	 {
	    		 System.out.println("It is not a triangle");
	    	 }
	     }
}
