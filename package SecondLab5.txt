package SecondLab;
//write a program to check leap year or not.
public class LeapYear {
	public static void main(String [] args)
    {
  	  int year=2024;
  	  if(year%4 == 0)
  	  {
  		  System.out.println("It is a leap year");
  	  }
  	  else
  	  {
  		  System.out.println("It is not a leap year");
  	  }
    }
}
